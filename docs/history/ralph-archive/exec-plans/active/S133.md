```python
import hashlib
import logging
import uuid
from pathlib import Path

import fitz  # PyMuPDF
from PIL import Image as PILImage

logger = logging.getLogger(__name__)

_MIN_WIDTH = 150
_MIN_HEIGHT = 100
_MAX_DIM = 4000

class ExtractedImage:
    """Value object returned from ImageExtractor; no DB references."""
    def __init__(
        self,
        page: int,
        index: int,
        width: int,
        height: int,
        content_hash: str,
        abs_path: Path,
        rel_path: str,
    ) -> None:
        self.id = str(uuid.uuid4())
        self.page = page
        self.index = index
        self.width = width
        self.height = height
        self.content_hash = content_hash
        self.abs_path = abs_path
        self.rel_path = rel_path


def extract_images_pdf(
    file_path: Path,
    images_dir: Path,
    doc_id: str,
) -> list[ExtractedImage]:
    """Extract images from a PDF file.

    Uses PyMuPDF page.get_images(full=True) + doc.extract_image(xref).
    Normalizes all images to PNG.
    Skips images outside the 150x100 to 4000x4000 bounds.
    Deduplicates via SHA-256 hash within this document.
    Returns list of ExtractedImage value objects (does NOT write to DB).
    """
    out_dir = images_dir / doc_id
    out_dir.mkdir(parents=True, exist_ok=True)
    results: list[ExtractedImage] = []
    seen_hashes: set[str] = set()

    doc = fitz.open(str(file_path))
    for page_idx in range(len(doc)):
        page = doc[page_idx]
        image_list = page.get_images(full=True)
        for img_idx, img_info in enumerate(image_list):
            xref = img_info[0]
            try:
                img_data = doc.extract_image(xref)
            except Exception as exc:
                logger.warning("PDF image extraction failed xref=%d page=%d: %s", xref, page_idx, exc)
                continue

            raw_bytes = img_data["image"]
            content_hash = hashlib.sha256(raw_bytes).hexdigest()
            if content_hash in seen_hashes:
                continue

            # Size check using PIL to avoid decoding invalid images
            try:
                from io import BytesIO
                img_pil = PILImage.open(BytesIO(raw_bytes))
                w, h = img_pil.size
            except Exception:
                continue

            if w < _MIN_WIDTH or h < _MIN_HEIGHT:
                continue  # decorative / too small

            if w > _MAX_DIM or h > _MAX_DIM:
                logger.warning(
                    "Skipping oversized image doc=%s page=%d index=%d size=%dx%d",
                    doc_id, page_idx, img_idx, w, h,
                )
                continue

            # Normalize to PNG and save
            rel_path = f"images/{doc_id}/{page_idx}_{img_idx}.png"
            abs_path = images_dir / doc_id / f"{page_idx}_{img_idx}.png"
            try:
                img_pil.save(str(abs_path), format="PNG")
            except Exception as exc:
                logger.warning("Could not save PNG doc=%s page=%d: %s", doc_id, page_idx, exc)
                continue

            seen_hashes.add(content_hash)
            results.append(
                ExtractedImage(
                    page=page_idx,
                    index=img_idx,
                    width=w,
                    height=h,
                    content_hash=content_hash,
                    abs_path=abs_path,
                    rel_path=rel_path,
                )
            )

    doc.close()
    logger.info(
        "PDF image extraction complete doc=%s pages=%d images=%d",
        doc_id, len(doc), len(results),
    )
    return results


def extract_images_epub(
    file_path: Path,
    images_dir: Path,
    doc_id: str,
) -> list[ExtractedImage]:
    """Extract images from an EPUB file.

    Uses ebooklib to iterate EpubImage items.
    Same size/dedup rules as PDF extraction.
    """
    import ebooklib  # noqa: PLC0415
    from ebooklib import epub  # noqa: PLC0415
    from io import BytesIO  # noqa: PLC0415

    out_dir = images_dir / doc_id
    out_dir.mkdir(parents=True, exist_ok=True)
    results: list[ExtractedImage] = []
    seen_hashes: set[str] = set()

    try:
        book = epub.read_epub(str(file_path), options={"ignore_ncx": True})
    except Exception as exc:
        logger.warning("EPUB open failed doc=%s: %s", doc_id, exc)
        return []

    img_idx = 0
    for item in book.get_items_of_type(ebooklib.ITEM_IMAGE):
        raw_bytes = item.get_content()
        content_hash = hashlib.sha256(raw_bytes).hexdigest()
        if content_hash in seen_hashes:
            continue

        try:
            img_pil = PILImage.open(BytesIO(raw_bytes))
            w, h = img_pil.size
        except Exception:
            continue

        if w < _MIN_WIDTH or h < _MIN_HEIGHT:
            continue

        if w > _MAX_DIM or h > _MAX_DIM:
            logger.warning(
                "Skipping oversized EPUB image doc=%s index=%d size=%dx%d",
                doc_id, img_idx, w, h,
            )
            continue

        rel_path = f"images/{doc_id}/epub_{img_idx}.png"
        abs_path = out_dir / f"epub_{img_idx}.png"
        try:
            img_pil.save(str(abs_path), format="PNG")
        except Exception as exc:
            logger.warning("Could not save EPUB PNG doc=%s: %s", doc_id, exc)
            continue

        seen_hashes.add(content_hash)
        results.append(
            ExtractedImage(
                page=0,  # EPUB has no page numbers
                index=img_idx,
                width=w,
                height=h,
                content_hash=content_hash,
                abs_path=abs_path,
                rel_path=rel_path,
            )
        )
        img_idx += 1

    logger.info("EPUB image extraction complete doc=%s images=%d", doc_id, len(results))
    return results
```

### Step 4 — Service: `EnrichmentQueueWorker`

File: `/Users/sethurama/DEV/LM/learning-mate/backend/app/services/enrichment_worker.py`

```python
import asyncio
import logging
import uuid
from collections.abc import Callable, Coroutine
from datetime import UTC, datetime
from typing import Any

from sqlalchemy import select, update

from app.database import get_session_factory
from app.models import DocumentModel, EnrichmentJobModel

logger = logging.getLogger(__name__)

# Type alias for job handler: (document_id, job_id) -> None (raises on failure)
JobHandler = Callable[[str, str], Coroutine[Any, Any, None]]


class EnrichmentQueueWorker:
    """Background worker that drains the enrichment_jobs table.

    Design:
    - One asyncio Task per document_id runs at a time.
    - Jobs for different documents run in parallel.
    - Jobs for the same document run sequentially (FIFO by created_at).
    - Handlers are registered per job_type via register().
    - poll_interval_s controls how often idle workers check for new jobs.
    """

    def __init__(self, poll_interval_s: float = 5.0) -> None:
        self._handlers: dict[str, JobHandler] = {}
        self._poll_interval_s = poll_interval_s
        self._running = False
        self._task: asyncio.Task | None = None
        # Strong reference set — prevents GC on running document tasks
        self._doc_tasks: set[asyncio.Task] = set()
        self._active_doc_ids: set[str] = set()

    def register(self, job_type: str, handler: JobHandler) -> None:
        """Register a handler for a job_type."""
        self._handlers[job_type] = handler
        logger.info("EnrichmentQueueWorker: registered handler for job_type=%s", job_type)

    async def start(self) -> None:
        """Start the background polling loop (called from lifespan)."""
        self._running = True
        self._task = asyncio.create_task(self._poll_loop())
        logger.info("EnrichmentQueueWorker: started")

    async def stop(self) -> None:
        """Stop the polling loop (called from lifespan shutdown)."""
        self._running = False
        if self._task:
            self._task.cancel()
            try:
                await self._task
            except asyncio.CancelledError:
                pass
        logger.info("EnrichmentQueueWorker: stopped")

    async def _poll_loop(self) -> None:
        while self._running:
            await self._dispatch_pending()
            await asyncio.sleep(self._poll_interval_s)

    async def _dispatch_pending(self) -> None:
        """Find pending jobs and launch per-document task runners for any doc not already running."""
        async with get_session_factory()() as session:
            result = await session.execute(
                select(EnrichmentJobModel.document_id)
                .where(EnrichmentJobModel.status == "pending")
                .distinct()
            )
            doc_ids = [row[0] for row in result.all()]

        for doc_id in doc_ids:
            if doc_id in self._active_doc_ids:
                continue  # Already running for this document
            self._active_doc_ids.add(doc_id)
            task = asyncio.create_task(self._run_for_document(doc_id))
            self._doc_tasks.add(task)
            task.add_done_callback(self._doc_tasks.discard)
            task.add_done_callback(lambda t, d=doc_id: self._active_doc_ids.discard(d))

    async def _run_for_document(self, document_id: str) -> None:
        """Process all pending jobs for one document sequentially (FIFO)."""
        while True:
            async with get_session_factory()() as session:
                result = await session.execute(
                    select(EnrichmentJobModel)
                    .where(
                        EnrichmentJobModel.document_id == document_id,
                        EnrichmentJobModel.status == "pending",
                    )
                    .order_by(EnrichmentJobModel.created_at)
                    .limit(1)
                )
                job = result.scalar_one_or_none()

            if job is None:
                break  # No more pending jobs for this document

            await self._run_job(job.id, job.document_id, job.job_type)

    async def _run_job(self, job_id: str, document_id: str, job_type: str) -> None:
        handler = self._handlers.get(job_type)

        async with get_session_factory()() as session:
            await session.execute(
                update(EnrichmentJobModel)
                .where(EnrichmentJobModel.id == job_id)
                .values(status="running", started_at=datetime.now(UTC))
            )
            await session.commit()

        logger.info(
            "EnrichmentQueueWorker: starting job job_id=%s job_type=%s doc=%s",
            job_id, job_type, document_id,
        )

        if handler is None:
            error_msg = f"No handler registered for job_type={job_type}"
            logger.warning(error_msg)
            async with get_session_factory()() as session:
                await session.execute(
                    update(EnrichmentJobModel)
                    .where(EnrichmentJobModel.id == job_id)
                    .values(
                        status="failed",
                        completed_at=datetime.now(UTC),
                        error_message=error_msg,
                    )
                )
                await session.commit()
            return

        try:
            await handler(document_id, job_id)
            async with get_session_factory()() as session:
                await session.execute(
                    update(EnrichmentJobModel)
                    .where(EnrichmentJobModel.id == job_id)
                    .values(status="done", completed_at=datetime.now(UTC))
                )
                await session.commit()
            logger.info(
                "EnrichmentQueueWorker: job done job_id=%s doc=%s", job_id, document_id
            )
        except Exception as exc:
            logger.warning(
                "EnrichmentQueueWorker: job failed job_id=%s doc=%s: %s",
                job_id, document_id, exc,
                exc_info=exc,
            )
            async with get_session_factory()() as session:
                await session.execute(
                    update(EnrichmentJobModel)
                    .where(EnrichmentJobModel.id == job_id)
                    .values(
                        status="failed",
                        completed_at=datetime.now(UTC),
                        error_message=str(exc),
                    )
                )
                await session.commit()


# Singleton worker instance (registered in lifespan)
_worker: EnrichmentQueueWorker | None = None


def get_enrichment_worker() -> EnrichmentQueueWorker:
    global _worker
    if _worker is None:
        _worker = EnrichmentQueueWorker()
    return _worker
```

### Step 5 — Service: Image extraction handler

The image extraction handler is defined inside `enrichment_worker.py` or as a module-level function in `image_extractor.py`. For clean separation, define it in `image_extractor.py` as a function `image_extract_handler(document_id: str, job_id: str) -> None` that:

1. Reads the `DocumentModel` to get `file_path` and `format`.
2. Computes `images_dir = Path(get_settings().DATA_DIR).expanduser() / "images"`.
3. Dispatches to `extract_images_pdf` or `extract_images_epub` based on format.
4. Finds the nearest preceding prose chunk for each extracted image: query `ChunkModel` for the document ordered by `page_number`; for each image, find the chunk with the highest `page_number <= image.page`.
5. Writes `ImageModel` rows — skipping rows where `content_hash` already exists for this document (dedup guard).
6. Sets `DocumentModel.stage = 'complete'` after all images are stored (if stage was 'enriching').

```python
async def image_extract_handler(document_id: str, job_id: str) -> None:
    """Enrichment handler for job_type='image_extract'.

    Called by EnrichmentQueueWorker for each image_extract job.
    Extracts images from the document file, stores ImageModel rows.
    Non-fatal: failures are caught by the worker and set job status='failed'.
    """
    from pathlib import Path  # noqa: PLC0415
    from sqlalchemy import select as _select  # noqa: PLC0415
    from sqlalchemy import update as _update  # noqa: PLC0415
    import uuid as _uuid  # noqa: PLC0415

    from app.config import get_settings  # noqa: PLC0415
    from app.database import get_session_factory  # noqa: PLC0415
    from app.models import ChunkModel, DocumentModel, ImageModel  # noqa: PLC0415

    settings = get_settings()
    images_dir = Path(settings.DATA_DIR).expanduser() / "images"
    images_dir.mkdir(parents=True, exist_ok=True)

    async with get_session_factory()() as session:
        doc_result = await session.execute(
            _select(DocumentModel).where(DocumentModel.id == document_id)
        )
        doc = doc_result.scalar_one_or_none()
        if doc is None:
            raise ValueError(f"Document not found: {document_id}")

        # Load existing hashes to deduplicate across re-extraction
        existing_hashes_result = await session.execute(
            _select(ImageModel.content_hash).where(ImageModel.document_id == document_id)
        )
        existing_hashes: set[str] = {row[0] for row in existing_hashes_result.all()}

        # Load chunks for nearest-chunk resolution (ordered by page_number)
        chunks_result = await session.execute(
            _select(ChunkModel.id, ChunkModel.page_number)
            .where(ChunkModel.document_id == document_id)
            .order_by(ChunkModel.page_number, ChunkModel.chunk_index)
        )
        chunks = chunks_result.all()  # list of (id, page_number)

    fmt = doc.format.lower()
    file_path = Path(doc.file_path)

    if fmt == "pdf":
        extracted = extract_images_pdf(file_path, images_dir, document_id)
    elif fmt == "epub":
        extracted = extract_images_epub(file_path, images_dir, document_id)
    else:
        logger.info(
            "image_extract_handler: format %s has no image extraction", fmt,
            extra={"doc_id": document_id},
        )
        return  # Text-only format — nothing to extract

    def _find_nearest_chunk(page: int) -> str | None:
        best_id: str | None = None
        for chunk_id, chunk_page in chunks:
            if chunk_page <= page:
                best_id = chunk_id
            else:
                break
        return best_id

    new_images: list[ImageModel] = []
    for img in extracted:
        if img.content_hash in existing_hashes:
            continue
        existing_hashes.add(img.content_hash)
        nearest_chunk = _find_nearest_chunk(img.page)
        new_images.append(
            ImageModel(
                id=img.id,
                document_id=document_id,
                chunk_id=nearest_chunk,
                page=img.page,
                path=img.rel_path,
                width=img.width,
                height=img.height,
                content_hash=img.content_hash,
            )
        )

    if new_images:
        async with get_session_factory()() as session:
            session.add_all(new_images)
            await session.commit()

    # Transition stage back to 'complete' (was set to 'enriching' by ingestion)
    async with get_session_factory()() as session:
        await session.execute(
            _update(DocumentModel)
            .where(
                DocumentModel.id == document_id,
                DocumentModel.stage == "enriching",
            )
            .values(stage="complete")
        )
        await session.commit()

    logger.info(
        "image_extract_handler: done doc=%s new_images=%d",
        document_id, len(new_images),
    )
```

```python
async def enrichment_enqueue_node(state: IngestionState) -> IngestionState:
    """Create an image_extract enrichment job and set stage='enriching'.

    Only runs for formats that may contain images (PDF, EPUB).
    For all other formats, transitions directly to complete.
    Non-fatal: enrichment failure does not prevent document from being usable.
    """
    import uuid as _uuid  # noqa: PLC0415
    from sqlalchemy import update as _update  # noqa: PLC0415
    from app.models import DocumentModel, EnrichmentJobModel  # noqa: PLC0415

    doc_id = state["document_id"]
    fmt = state.get("format", "").lower()
    _IMAGE_FORMATS = {"pdf", "epub"}

    if fmt not in _IMAGE_FORMATS:
        # Not a format with images — stage stays 'complete' (set by entity_extract_node)
        logger.info(
            "enrichment_enqueue_node: skipping (format=%s has no images)", fmt,
            extra={"doc_id": doc_id},
        )
        return state

    try:
        job_id = str(_uuid.uuid4())
        async with get_session_factory()() as session:
            session.add(EnrichmentJobModel(
                id=job_id,
                document_id=doc_id,
                job_type="image_extract",
                status="pending",
            ))
            await session.execute(
                _update(DocumentModel)
                .where(DocumentModel.id == doc_id)
                .values(stage="enriching")
            )
            await session.commit()

        # Notify the worker that there is a new job (avoid waiting for next poll)
        from app.services.enrichment_worker import get_enrichment_worker  # noqa: PLC0415
        worker = get_enrichment_worker()
        # Fire-and-forget dispatch for this document
        task = asyncio.create_task(worker._dispatch_pending())
        _background_tasks.add(task)
        task.add_done_callback(_background_tasks.discard)

        logger.info(
            "enrichment_enqueue_node: enqueued image_extract job=%s doc=%s",
            job_id, doc_id,
        )
    except Exception as exc:
        logger.warning(
            "enrichment_enqueue_node failed (non-fatal): %s",
            exc,
            extra={"doc_id": doc_id},
        )
    return state
```

```python
async def get_document_images(
    document_id: str,
    page: int = 1,
    page_size: int = 20,

    """Return paginated image list for a document.

    Returns empty list (not 404) for text-only documents.
    Returns 404 if the document does not exist.

            select(DocumentModel.id).where(DocumentModel.id == document_id)

            select(ImageModel)

            .order_by(ImageModel.page, ImageModel.created_at)

    total = len(all_images)
    start = (page - 1) * page_size
    page_images = all_images[start : start + page_size]
    return ImageListResponse(
        items=[
            ImageItem(
                id=img.id,
                document_id=img.document_id,
                chunk_id=img.chunk_id,
                page=img.page,
                path=img.path,
                width=img.width,
                height=img.height,
                content_hash=img.content_hash,
                image_type=img.image_type,
                description=img.description,
                created_at=img.created_at.isoformat(),
            )
            for img in page_images
        ],
        total=total,
        page=page,
        page_size=page_size,
    )
```

## Risk Flags

- **Risk: PyMuPDF `page.get_images()` returns duplicate xrefs for the same image** — mitigation: the SHA-256 dedup set `seen_hashes` inside `extract_images_pdf` prevents storing duplicate image files from the same extraction pass. The `ImageModel.UniqueConstraint("document_id", "content_hash")` further prevents DB-level duplicates across re-runs.

- **Risk: `ebooklib` `read_epub` emits deprecation warnings that may appear in pytest output** — mitigation: pass `options={"ignore_ncx": True}` to suppress the most common warning. These are warnings only and do not affect functionality.

- **Risk: `enrichment_enqueue_node` wiring breaks the graph edge order** — current terminal chain is `summarize -> END`. The new chain is `summarize -> enrichment_enqueue -> END`. The `error_finalize` path is unaffected (it still goes `error_finalize -> END`). Carefully update only the `builder.add_edge("summarize", END)` line — do not touch any other edge. Mitigation: `STAGE_PROGRESS` gains `"enriching": 95` so the status poll endpoint returns a sensible progress value.

- **Risk: TD-010 (no FK CASCADE)** — `EnrichmentJobModel` and `ImageModel` rows must be explicitly deleted in both `delete_document` and `bulk_delete_documents`. The plan includes this in Step 8. Failure to add would orphan rows and leave orphaned PNG files in `DATA/images/{doc_id}/`.

- **Risk: `DATA/images/` directory not created at startup** — mitigation: add `"images"` to the subdirectory creation loop in `lifespan` (Step 9). Also, `extract_images_pdf` and `extract_images_epub` call `images_dir.mkdir(parents=True, exist_ok=True)` before writing, so the first extraction is safe even if startup was missed.

- **Risk: `image_extract_handler` called for non-PDF/EPUB formats** — mitigation: the handler checks `fmt` at the top and returns early for unsupported formats. The `enrichment_enqueue_node` also only creates the job for `_IMAGE_FORMATS = {"pdf", "epub"}`. Both guards must remain in sync.

- **Risk: Large PDFs (10,000+ pages) may generate hundreds of images, consuming significant disk** — mitigation: the `_MAX_DIM = 4000` filter and minimum size filter together discard most non-content images. Full-page scanned images (4001+ px) are skipped with a warning. For books with many embedded figures, disk usage is expected and acceptable (this is the feature's intent).

- **Risk: Frontend `enrichment_status` field missing from existing cached API responses** — mitigation: the field is declared `Optional[str] = None` in `DocumentListItem` (Pydantic), so older documents without enrichment jobs return `null`. The TypeScript type is `string | null` and the badge renders only when non-null. No breaking change.

- **Risk: `EnrichmentQueueWorker` polling interval creates a 5-second lag before jobs start** — mitigation: `enrichment_enqueue_node` explicitly calls `worker._dispatch_pending()` as a fire-and-forget task immediately after creating the job, so the latency is minimal in practice. The 5-second poll is only a fallback for jobs that missed the immediate dispatch.

- **Risk: Worker `_active_doc_ids` set may retain stale entries if a document task raises an unhandled exception** — mitigation: `_active_doc_ids.discard(d)` is added as a `done_callback` on the task, so it is always removed regardless of success or failure.
