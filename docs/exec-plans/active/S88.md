def _hit_rate_at_k(results: list[str], hints: list[str], k: int = 3) -> float:
    """Compute HR@k: fraction of hints where a hit appears in top-k results."""
    if not hints:
        return 0.0
    hits = 0
    per_hint_results = _group_by_hint(results, hints, k)
    for hint, chunks in per_hint_results:
        if any(hint.lower()[:80] in c.lower() for c in chunks):
            hits += 1
    return hits / len(hints)

@pytest.mark.slow
@pytest.mark.asyncio
async def test_alice_factual_retrieval_hr3(all_books_ingested):
    """HR@3 >= 0.6 on 5 Alice factual questions using the hybrid retriever."""
    from app.services.retriever import get_retriever

    alice_doc_id = all_books_ingested["Alice in Wonderland"]["doc_id"]
    retriever = get_retriever()

    factual_qs = [
        ("What label was on the bottle Alice found?", "DRINK ME"),
        ("What advice did the Caterpillar give Alice about the mushroom?",
         "One side will make you grow taller"),
        ("What were the balls and mallets in the Queen's croquet game?",
         "the balls were live hedgehogs, the mallets live flamingoes"),
        ("What did the Cheshire Cat say about everyone in Wonderland?",
         "we're all mad here"),
        ("What did the Dodo declare the winner of the Caucus-race?",
         "_Everybody_ has won, and all must have prizes"),
    ]

    hits = 0
    for question, hint in factual_qs:
        chunks = await retriever.search(
            query=question,
            document_ids=[alice_doc_id],
            k=3,
        )
        texts = [c.text for c in chunks]
        if any(hint.lower()[:80] in t.lower() for t in texts):
            hits += 1

    hr3 = hits / len(factual_qs)
    assert hr3 >= 0.6, (
        f"Alice HR@3={hr3:.2f} < 0.60 on {len(factual_qs)} factual questions"
    )

async def test_alice_executive_summary_quality(all_books_ingested):
    """Alice executive summary: no 3-line numbered-list pattern; mentions key terms."""
    import app.repos.summary_repo as summary_repo_module

    alice_doc_id = all_books_ingested["Alice in Wonderland"]["doc_id"]
    repo = summary_repo_module.SummaryRepo()

    text = exec_summary.summary or ""

    # Check no numbered-list pattern (3 or more consecutive lines starting 1. 2. 3.)
    lines = text.splitlines()
    numbered_consecutive = 0
    for line in lines:
        if re.match(r"^\s*\d+\.\s", line):
            numbered_consecutive += 1
        else:
            numbered_consecutive = 0
        assert numbered_consecutive < 3, (
            "Alice executive summary looks like a passage-list (3+ numbered lines):\n"
            + text[:400]
        )

    # Must mention key characters and themes
    text_lower = text.lower()
    required_terms = ["alice", "wonderland", "queen", "rabbit", "mad"]
    missing = [t for t in required_terms if t not in text_lower]
    assert not missing, (
        f"Alice executive summary missing terms {missing}:\n{text[:400]}"
    )

## Risk Flags

- Risk: `summary_repo` may not expose `get_section_summaries` or `get_executive_summary`
  by those exact names. Mitigation: Step 4 requires reading the repo file before writing
  the test. Adjust method calls to match actual API before committing.

- Risk: `retriever.search()` may use different parameter names (`doc_ids` vs
  `document_ids`, `limit` vs `k`, or may not accept `None` for no-filter cross-book
  retrieval). Mitigation: Step 5 requires reading the retriever service signature first.

- Risk: `graph_module.get_graph_service()` may expose entity data via a different method
  (e.g., `.get_entities()`, `.query_entities(doc_id)`). Mitigation: Step 6 requires
  reading graph.py before writing entity tests.

- Risk: GLiNER may canonicalize "Ulysses" instead of "Odysseus" for the Odyssey since the
  Butler translation uses "Ulysses" throughout. Mitigation: the Odyssey entity test checks
  for `"odysseus" in n` as a substring -- adjust to also check for `"ulysses"` as an
  alternative if the test fails at validation time, or add "Ulysses" to the required-terms
  check.

- Risk: The Odyssey is very long (~100k words). Section summary count may be very high but
  the test only requires >= 10, which is conservative. The ingest time budget for The
  Odyssey is 1800 seconds (30 minutes) -- the conftest already enforces this.

- Risk: `context_hint` matching uses first 80 characters lowercased. All hints chosen are
  shorter than 80 unique characters or the first 80 characters are distinctive enough to
  be unambiguous. If a hint contains markdown formatting (underscores), the hint was
  copied verbatim from the source including those underscores, since `compute_hit_rate_5`
  lowercases but does not strip markdown. Chunks may or may not contain the underscore
  markup depending on how the text extractor normalizes the .txt source.
  Mitigation: prefer context_hints that do not contain underscore markup. All 40 entries
  above were chosen to avoid underscore-heavy phrases where possible; the few that use
  underscores (e.g., "DRINK ME") are all-caps labels that are unambiguous.

- Risk: `test_cross_book_retrieval` passes `document_ids=None` to `retriever.search()`.
  If the retriever signature does not accept `None` (requires a list), this will throw a
  TypeError at test time. Mitigation: Step 5 verifies the signature; if `None` is not
  accepted, pass all three doc_ids as a list and assert >= 2 distinct values appear.

- Risk: no `@pytest.mark.asyncio` import unless `pytest-asyncio` is installed and
  configured. Check `pyproject.toml` for `asyncio_mode = "auto"` in pytest config -- if
  present, the decorator is not needed and can be omitted. If not present, each async
  test needs the decorator. Mitigation: check existing test files for the pattern used.
